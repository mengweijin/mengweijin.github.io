    # 解决方案

    由于资金有限，买高配置的服务器不现实。那么在不增加物理内存的情况下，启用swap交换区作为虚拟内存是一个不错的选择。

    注意：如果使用 Kubernetes 的话，此方法不可行。安装 Kubernetes 必须关闭 swap。

    参考文档：Linux 启用swap交换区作为虚拟内存 (opens new window)

    # 增加交换内存 swap

    # 查看系统当前的分区情况
    [root@localhost ~]# free -h
              total        used        free      shared  buff/cache   available
Mem:           972M        202M        661M        7.6M        107M        642M
Swap:          0           0           0

# 可以看出，当前是没有启用 swap 交换区的。

    # 创建用于交换分区的文件
    # 先创建一个 swap 文件存放的文件夹
mkdir /home/swap

# 创建8GB的交换内存（一般大小设为为物理内存的两倍即可）
# (bs*count=文件大小) 会在 /home/swap/下创建名为 data-1 的文件（交换内存文件）
# 创建需要一点时间，需要耐心等待
dd if=/dev/zero of=/home/swap/data-1 bs=1M count=8192 

# 把这个文件设置为交换分区文件
mkswap /home/swap/data-1

# 修改文件权限为600
chmod 600 /home/swap/data-1

# 启用交换分区文件
swapon /home/swap/data-1

# 开机时自启用。若要想使开机时自启用，则需修改文件/etc/fstab 添加一行
vi /etc/fstab
# 添加以下内容：
/home/swap/data-1 swap swap defaults 0 0

    # 使用 crontab 让应用自动重启

    注意：脚本中 “source /etc/profile” 是因为 crontab 执行脚本无法取到环境变量，导致进程启动不起来，所以这个必须加上。

    start.sh 和 monitor.sh 存放于 /root 目录下

    # 假如启动java进程脚本为：start.sh
    #! /bin/sh
source /etc/profile
 
kill -9 `ps -ef|grep 'project.jar'|grep -v grep|awk '{print $2}'`
nohup java -jar project.jar >project.log 2>&1 &
tail -f project.log

    # 监控shell脚本：monitor.sh
    #! /bin/sh
source /etc/profile
cd /root

procnum=`ps -ef|grep project.jar|grep -v grep|wc -l`
if [ $procnum -eq 0 ]
then
    echo `date +%Y-%m-%d` `date +%H:%M:%S` "restart service" >>/root/logs/restart.log
    ./start.sh
fi
